Error :: File


Description:
Retrieves the script file name that the soft error occurred in.
Syntax:
Error.File(Index as Numeric = 0) as String
Parameters:
  • Index as Numeric
    • The index of the soft error in the error stack.
Example:
<%...%>
<%

Sleep(-100) ;This will produce the error "Sleep time must be greater than zero"

If (Error.Count()) ...
If(Error.Count())
    Print(Error.Count() & " error(s) occured:")

    For (iIndex as Numeric < Error.Count()) ...
    For(iIndex as Numeric < Error.Count())
        Print(Tab & iIndex + 1 & ") " & Error.File(iIndex) & CrLf & Tab & Tab & Error.Text(iIndex) & " on line " & Error.Line(iIndex)) 
        iIndex++
    Next

    Error.Clear()
Else
    Print("No errors occured.")
End If
