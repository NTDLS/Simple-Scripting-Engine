XML :: Writer :: Length


Description:
Retrieves the full length of the text contained in the XML Writer.
Syntax:
XML.Writer.Length() as Numeric
Parameters:
    Example:
    <%...%>
    <%

    var people as XML.Writer
    people.SetName("People")

    For (i as Numeric = 0 to 10) ...
    for(i as Numeric = 0 to 10)
        var person as XML.Writer
        
        person.SetName("Person")
        person.Add("Id", i)
        person.Add("FirstName", "Steve_" & i)
        person.Add("LastName", "Jobs_" & i)

        people.AddXml(@person)    
    Next
        
    Print("Length: " & people.Length() & ", Text: [" & people.Text() & "]")

    %>
