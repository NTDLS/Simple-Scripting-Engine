Error :: Count


Description:
Retrieves the current count of soft errors from the soft error stack.
Syntax:
Error.Count() as Numeric
Parameters:
    Example:
    <%...%>
    <%

    Sleep(-100) ;This will produce the error "Sleep time must be greater than zero"

    If (Error.Count()) ...
    If(Error.Count())
        Print(Error.Count() & " error(s) occured:")

        For (iIndex as Numeric < Error.Count()) ...
        For(iIndex as Numeric < Error.Count())
            Print(Tab & iIndex + 1 & "): " & Error.Text(iIndex) & " on line " & Error.Line(iIndex))
            iIndex++
        Next

        Error.Clear()
    Else
        Print("No errors occured.")
    End If

    %>
