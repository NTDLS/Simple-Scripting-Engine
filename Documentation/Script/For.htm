For


Description:
The FOR statement allows you to loop through a given number of iterations, provides the convenience of optionally declaring an internal iterator and automatically incrementing or decrementing it by an optional stepping (with a default stepping of 1 or -1 respectively).
Example:
<%...%>
<%

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
var outsideCount as Numeric = 0

Print("Loop forward by 1 using external variable.")
For (outsideCount = 0 to 5) ...
For(outsideCount = 0 to 5)
    Print(outsideCount)
Next

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Print("Loop forward by 1 using internal variable.")
For (count as Numeric = 0 to 5) ...
For(count as Numeric = 0 to 5)
    Print(count)
Next

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Print("Loop forwards by 10 using internal variable.")
For (count as Numeric = 0 to 100) ...(10)
For(count as Numeric = 0 to 100)
    Print(count)
Next(10)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Print("Loop backwards by -10 using internal variable.")
For (count as Numeric = 100 to 0) ...(-10)
For(count as Numeric = 100 to 0)
    Print(count)
Next(-10)


%>
