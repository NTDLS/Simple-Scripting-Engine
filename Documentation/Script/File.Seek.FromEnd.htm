File :: Seek :: FromEnd


Description:
Moves the file read or write cursor a specified number of bytes from the end of a file.
Syntax:
File.Seek.FromEnd(Distance as Numeric = 0) as Numeric
Parameters:
  • Distance as Numeric
    • Number of bytes to move the file cursor realitive to the end of the file.
Remarks:
Returns the new position of the cursor in the file. You can use this function to move the file cursor to the beginning of the file by passing a distance of zero.
Example:
<%...%>
<%

Var txtFile As File

If (txtFile.Open("C:\\Test.txt", FileAccess.Write)) ...
If(txtFile.Open("C:\\Test.txt", FileAccess.Write))
    Var BytesWritten as Numeric = txtFile.Write("This is the contents of the file!")
    Print("Write: " & BytesWritten & " bytes.")
    txtFile.Close()

    If (txtFile.Open("C:\\Test.txt", FileAccess.Read)) ...
    If(txtFile.Open("C:\\Test.txt", FileAccess.Read))
        txtFile.Seek.FromEnd(-10) ;Seek from the end of the file 
        Print("Read: [" & txtFile.Read() & "]")
        txtFile.Close()
    End If
End If

File.Delete("C:\\Test.txt")
