SQL :: Connection :: Execute


Description:
Executes a statement against the SQL server, binds results to a record set if one is passed to the function – otherwise no data is returned from the query.
Syntax:
SQL.Connection.Execute(SQLText as String, RecordSet as SQL.RecordSet = -1)
Parameters:
  • SQLText as String
    • SQL statement to be executed by SQL server
  • RecordSet as SQL.RecordSet
    • Optional pointer to a record set object to bind the result set to. Any results, if any, will be discarded if this parameter is not passed.
Example:
<%...%>
<%

var sqlConn as SQL.Connection
var recordSet as SQL.RecordSet

sqlConn.Connect("(local)")

sqlConn.Execute("SELECT TOP 10 id, name, xtype FROM sysobjects", @recordSet)

print("Index of Name A: " recordSet.Columns.Index("name"))
print("Name of Index 0: " recordSet.Columns.Name(2))

Print("")

;All Columns:
For (i as Numeric to recordSet.Columns.Count()) ...
For(i as Numeric to recordSet.Columns.Count())
    Prints(recordSet.Columns.Name(i) & "\t")
Next

Print("")

;All Rows:
While (recordSet.Fetch()) ...
While(recordSet.Fetch())
    var sRow as String

    For (i as Numeric to recordSet.Columns.Count()) ...
    For(i as Numeric to recordSet.Columns.Count())
        sRow.Append(recordSet.Value(i) & "\t")
    Next
    Print(sRow)
WEnd

recordSet.Close();
sqlConn.Close();

%>
