<%

var beforeTime as Numeric

var textString as String = "This is a test string."

''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
'Test 1: Check length of string and perform math at each iteration.
''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
beforeTime = TickCount()

For (i As Numeric = 0 to textString.Length() * 1000) ...
For(i As Numeric = 0 to textString.Length() * 1000)
    'Do something
Next

Print("Duration: " & FormatNumeric(TickCount() - beforeTime) & "ms")

''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
'Test 2: Check length of string and perform math once.
''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
beforeTime = TickCount()

var targetValue as Numeric = textString.Length() * 1000

For (i As Numeric = 0 to targetValue) ...
For(i As Numeric = 0 to targetValue)
    'Do something
Next

Print("Duration: " & FormatNumeric(TickCount() - beforeTime) & "ms")

%>
